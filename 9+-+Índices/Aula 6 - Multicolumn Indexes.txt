// Aula 6 - Multicolumn Indexes

https://www.postgresql.org/docs/current/indexes-multicolumn.html

=# CREATE TABLE pessoas(
    id INT GENERATED BY DEFAULT AS IDENTITY,
    primeiro_nome VARCHAR(50) NOT NULL,
    ultimo_nome VARCHAR(50) NOT NULL
);

=# SELECT * FROM pessoas WHERE ultimo_nome = 'Adams';

=# CREATE INDEX idx_pessoas_nomes ON pessoas (ultimo_nome, primeiro_nome);

=# EXPLAIN SELECT * FROM pessoas WHERE ultimo_nome = 'Adams';

=# SELECT * FROM pessoas WHERE ultimo_nome = 'Adams' AND primeiro_nome = 'Lou';

=# EXPLAIN SELECT * FROM pessoas WHERE ultimo_nome = 'Adams' AND primeiro_nome = 'Lou';

=# EXPLAIN SELECT * FROM pessoas WHERE primeiro_nome = 'Lou';
