=# select c.*,p.category,p.title from categories c left join posts p on c.id=p.category;

=# select * from categories c where c.id not in (select category from posts);

=# select * from categories c where not exists (select 1 from posts where category=c.id);

=# select c.*,p.category from categories c left join posts p on p.category=c.id;

=# select c.* from categories c left join posts p on p.category=c.id where p.category is null;

=# select c.*,p.category,p.title from posts p right join categories c on c.id=p.category;
